def get_all_operation(op1, op2):
    new_op = set()
    
    new_op.add(op1 + op2)
    new_op.add(op1 - op2)
    if op2 != 0:
        new_op.add(op1 // op2)
    new_op.add(op1 * op2)
    
    return new_op

def solution(N, number):
    answer = -1
